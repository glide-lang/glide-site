router.glide 
import stdlib::fs::*;
import stdlib::http::*;

fn handle(req: *HttpRequest) -> HttpResponse {
    if req.path.eq("/")         { return home(req); }
    if req.path.eq("/tutorial") { return tutorial(req); }
    if req.path.eq("/examples") { return examples(req); }
    return HttpResponse::not_found()
                   .body(fs_read("static/404.html"));
}
